$('input[name="opinion1"]').on('click', function() {
  if(this.value === '1') {
    $('#myOpinion1').empty().append('<h3>In general, AngularJS is straightforward.</h3>');
  } else {
    $('#myOpinion1').empty().append('<h3>Oh. no! AngularJS is so obscure.</h3>');
  }
});
