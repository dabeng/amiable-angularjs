ngIf

In general, AngularJS is straightforward.

Oh. no! AngularJS is so obscure.

html source code

  
<div ng-controller="NgIfCtrl">
  <label style="margin-right:20px;"> 
    <input type="radio" ng-model="opinion" ng-value="true"> I like AngularJS
  </label>
  <label>
    <input type="radio" ng-model="opinion" ng-value="false"> I dont't like it
  </label>
  <h3 ng-if="opinion">In general, AngularJS is straightforward.</h3>
  <h3 ng-if="!opinion">Oh. no! AngularJS is so obscure.</h3>
</div>

javascript source code

  
var switchDomApp = angular.module('switchDomApp', []);
switchDomApp.controller('NgIfCtrl', function($scope) {
  $scope.opinion = true;
});

html source code

  
<div ng-controller="NgSwitchCtrl">
  <label style="margin-right:20px;">
    <input type="radio" ng-model="opinion" value="0"> Neutral
  </label>
  <label style="margin-right:20px;"> 
    <input type="radio" ng-model="opinion" value="1"> Advocate
  </label>
  <label style="margin-right:20px;">
    <input type="radio" ng-model="opinion" value="2"> Opponent
  </label>
  <label>
    <input type="radio" ng-model="opinion" value="3">  Somebody
  </label>
  <div ng-switch on="opinion">
    <div ng-switch-when="1">
      <p class="bg-success diff-opinion">AngularJS comes from Google, I trust Google, so I trust AngularJS.</p>
      <p class="bg-info diff-opinion">It's so powerfull, I love it.</p>
      <p class="bg-warning diff-opinion">I love its elegant grammar.</p>
    </div>
    <div ng-switch-when="2">
      <p class="bg-danger diff-opinion">Oh. no! AngularJS is so obscure. It's my nightmare.</p>
      <p class="bg-warning diff-opinion">AngularJS is too bloated. It's not my type.</p>
    </div>
    <div ng-switch-when="3">
      <p class="bg-primary diff-opinion">Hey, dude! What's AngularJS?</p>
    </div>
    <div ng-switch-default>
      <p class="bg-success diff-opinion">You know I'm backend guy, I don't care.</p>
    </div>
  </div>
</div>

javascript source code

  
var switchDomApp = angular.module('switchDomApp', []);
switchDomApp.controller('NgSwitchCtrl', function($scope) {
  $scope.opinion = '1';
});
