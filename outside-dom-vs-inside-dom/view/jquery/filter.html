$('#keyword').on('keyup', function(event) {
  var candidates = $('#candidate-list').children();
  for (var i = 0; i < candidates.length; i++) {
    var item = candidates.eq(i);
    if ((new RegExp(event.target.value, 'i')).test(item.text())) {
      item.show();
    } else {
      item.hide();
    }
  }
});
